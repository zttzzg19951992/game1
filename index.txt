function shuffleArray(arr) {
  const newArr = [...arr];
  for (let i = newArr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [newArr[i], newArr[j]] = [newArr[j], newArr[i]];
  }
  return newArr;
}

// 固定每题4分，满分100
const perScore = 4;
const totalFullScore = 100;

let currentGameIndex = 0, q = 0, score = 0;
let curGame, curQ;

const gameTitle = document.getElementById("gameTitle");
const qBox = document.getElementById("qBox");
const ansBox = document.getElementById("ansBox");
const nextBtn = document.getElementById("nextBtn");
const scoreEl = document.getElementById("score");
const totalScoreEl = document.getElementById("totalScore");
const progressEl = document.getElementById("progress");
const tabBtns = document.querySelectorAll(".tab-btn");

nextBtn.innerText = "下一题";
totalScoreEl.innerText = totalFullScore;

function switchGame(index) {
  currentGameIndex = index;
  q = 0;
  tabBtns.forEach((btn, i) => btn.classList.toggle("active", i === index));
  loadGame();
}

function loadGame() {
  curGame = games[currentGameIndex];
  if (currentGameIndex === 1) {
    curGame.list = shuffleArray(curGame.list);
  }
  gameTitle.innerText = curGame.category + "挑战";
  nextBtn.disabled = true;
  loadQuestion();
}

function loadQuestion() {
  if (q >= curGame.list.length) {
    gameTitle.innerText = "? 本分类完成！";
    qBox.innerHTML = `<p>请选择其他分类继续挑战</p>`;
    ansBox.style.display = "none";
    nextBtn.disabled = true;
    nextBtn.innerText = "已完成";
    updateProgress();
    return;
  }
  curQ = curGame.list[q];
  updateProgress();
  ansBox.style.display = "none";
  nextBtn.disabled = true;
  nextBtn.innerText = "下一题";

  if (curQ.type === "fill") {
    renderSingleFill();
  } else {
    renderChoice();
  }
}

// 单选题
function renderChoice() {
  qBox.innerHTML = `<div class="question">${curQ.q}</div>`;
  const wrap = document.createElement("div");
  curQ.opts.forEach((txt, i) => {
    const btn = document.createElement("button");
    btn.className = "btn";
    btn.innerText = txt;
    btn.onclick = () => selectAnswer(i, btn);
    wrap.appendChild(btn);
  });
  qBox.appendChild(wrap);
}

function selectAnswer(index, btn) {
  const all = qBox.querySelectorAll(".btn");
  all.forEach(b => { b.disabled = true; });
  if (index === curQ.ans) {
    btn.classList.add("ok");
    score += perScore;
    ansBox.className = "answer-box ok";
    ansBox.innerHTML = "? 正确！+" + perScore + "分<br>" + curQ.exp;
  } else {
    btn.classList.add("err");
    all[curQ.ans].classList.add("ok");
    ansBox.className = "answer-box err";
    ansBox.innerHTML = "? 错误<br>正确答案：" + curQ.opts[curQ.ans] + "<br>" + curQ.exp;
  }
  ansBox.style.display = "block";
  nextBtn.disabled = false;
  updateScore();
}

// 填空题：新增答案文字+字数提示+提交按钮，取消输入实时判断
function renderSingleFill() {
  qBox.innerHTML = `
    <div class="question">${curQ.q}</div>
    <div style="text-align:center; margin-top:10px;">
      <div class="tip-word">温馨提示：本题答案为${curQ.wordNum}个字</div>
      <span>答案：</span>
      <input class="poem-input" type="text" maxlength="${curQ.wordNum}" id="fillInput">
      <button class="btn submit-fill" id="submitFillBtn">提交答案</button>
    </div>
  `;
  const submitBtn = document.getElementById("submitFillBtn");
  submitBtn.onclick = checkSingleFill;
}

// 点击提交才判分
function checkSingleFill() {
  const input = document.getElementById("fillInput");
  const val = input.value.trim();
  if (!val) {
    alert("请填写答案后再提交！");
    return;
  }

  if (val === curQ.ans) {
    ansBox.className = "answer-box ok";
    ansBox.innerHTML = "? 正确！+" + perScore + "分<br>" + curQ.exp;
    score += perScore;
  } else {
    ansBox.className = "answer-box err";
    ansBox.innerHTML = "? 错误<br>正确答案：" + curQ.ans + "<br>" + curQ.exp;
  }
  ansBox.style.display = "block";
  nextBtn.disabled = false;
  updateScore();
  //提交后锁定输入和提交按钮
  input.disabled = true;
  document.getElementById("submitFillBtn").disabled = true;
}

// 下一步
nextBtn.onclick = () => {
  q++;
  loadQuestion();
};

function updateProgress() {
  progressEl.innerText = `${q + 1}/${curGame.list.length}`;
}

function updateScore() {
  scoreEl.innerText = score;
}

loadGame();
